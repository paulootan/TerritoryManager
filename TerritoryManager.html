function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action || 'getAll'; const result = {}; if (action === 'getAll') { result.territories = sheetToJSON(ss.getSheetByName('Territories')); result.assignments = sheetToJSON(ss.getSheetByName('Assignments')); result.comments = sheetToJSON(ss.getSheetByName('Comments')); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'addAssignment') { const sheet = ss.getSheetByName('Assignments'); const id = sheet.getLastRow(); sheet.appendRow([id, data.territory_id, data.assigned_to, data.date_assigned, data.date_completed || '', data.service_year]); updateLastCompleted(ss, data.territory_id, data.date_completed); } if (action === 'completeAssignment') { updateAssignmentRow(ss, data.assignment_id, data.date_completed); updateLastCompleted(ss, data.territory_id, data.date_completed); } if (action === 'addComment') { const sheet = ss.getSheetByName('Comments'); const id = sheet.getLastRow(); sheet.appendRow([id, data.territory_id, data.assignment_id, data.comment_text, data.comment_date]); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function sheetToJSON(sheet) { if (!sheet) return []; const data = sheet.getDataRange().getValues(); const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function updateLastCompleted(ss, tid, date) { if (!date) return; const sheet = ss.getSheetByName('Territories'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == tid) { sheet.getRange(i+1, 2).setValue(date); return; } } sheet.appendRow([tid, date]); } function updateAssignmentRow(ss, aid, date) { const sheet = ss.getSheetByName('Assignments'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == aid) { sheet.getRange(i+1, 5).setValue(date); return; } } }
